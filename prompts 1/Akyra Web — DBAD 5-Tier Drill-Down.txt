export async function deleteOrganization(orgId: string): Promise<boolean> {
  const { error } = await supabase.from("organizations").delete().eq("id", orgId)
  if (error) {
    console.error("deleteOrganization failed:", error.message)
    return false
  }
  return true
}


Update fetchDistrictsForRegion to filter by region_id:

export async function fetchDistrictsForRegion(regionId: string) {
  const { data, error } = await supabase
    .from("districts")
    .select("id, name, org_id, district_manager_id, stores(id)")
    .eq("region_id", regionId)
    .order("name")

  if (error || !data) return []

  return data.map((d: any) => ({
    id: d.id,
    name: d.name,
    orgId: d.org_id,
    districtManagerId: d.district_manager_id,
    storeCount: d.stores?.length ?? 0,
  }))
}


Update createDistrict to accept and insert regionId:

export async function createDistrict(
  orgId: string,
  regionId: string,
  name: string
): Promise<string | null> {
  const { data, error } = await supabase
    .from("districts")
    .insert({ org_id: orgId, region_id: regionId, name })
    .select("id")
    .maybeSingle()

  if (error) {
    console.error("createDistrict failed:", error.message)
    return null
  }
  return data?.id ?? null
}
